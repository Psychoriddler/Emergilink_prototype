        // --- Navigation and Routing ---

        /**
         * Handles navigation between pages.
         * @param {string} targetPage - The key of the page to navigate to.
         */
        function navigate(targetPage) {
            if (currentPage === targetPage) return; // Prevent unnecessary re-render

            currentPage = targetPage;
            
            // Render content
            contentDiv.innerHTML = pages[targetPage];
            
            // Re-initialize Lucide icons for the new content
            lucide.createIcons();
            
            // Update active state of navigation bar
            document.querySelectorAll('.nav-icon').forEach(btn => {
                const navId = btn.id.replace('nav-', '');
                if (navId === targetPage) {
                    btn.classList.add('text-pink-500');
                    btn.classList.remove('text-gray-500');
                } else {
                    btn.classList.remove('text-pink-500');
                    btn.classList.add('text-gray-500');
                }
            });

            // Ensure we scroll to the top of the new page content
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
